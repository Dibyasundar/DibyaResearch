  1. Write a function in C to swap two variables by using pass-by-reference.

Answer

#include<stdio.h>
void swap(int *p, int *q)
{
  int r;
  r=*p;
  *p=*q;
  *q=r;
}
int main()
{
  int a=5;
  int b=6;
  printf("\nValue of a before swapping : %d",a);
  printf("\nValue of b before swapping : %d",b);

  printf("\n\n Calling the swap function\n\n");
  swap(&a,&b);
  printf("\nValue of a after swapping : %d",a);
  printf("\nValue of b after swapping : %d\n",b);

  return(0);
}

  1. Write a program in C to store n elements in an array and print the elements using pointer.

Answer

#include<stdio.h>
#include<stdlib.h>
int main()
{
  int a[10],i;
  int *b;
  printf("Enter the elements in array");
  for(i=0;i<10;i++)
  {
    scanf("%d",&a[i]);  
  }
  b=a;
  for(i=0;i<10;i++)
  {
    printf("%d ",*(b+i));
  }
  return(0);
}

  1. Write a C program that uses a function that can return multiple value of different data type using pointer and without using structure.

Answer

#include<stdio.h>
#include<stdlib.h>
void** ret()
{
    void **p;
    int a=50;
    char b='c';
    float c=1.2;
    p = (void **)malloc(3*sizeof(void));
    p[0]=&a;
    p[1]=&b;
    p[2]=&c;
    return(p);
}

int main()
{
    void **p;
    p=ret();
    printf("%d\n",*(int *)p[0]);
    printf("%c\n",*(char *)p[1]);

    return(0);
}

  1. Write a C Program to list all files and sub-directories in a directory (Hint: use #include <dirent.h>)

Answer

#include <stdio.h>
#include <dirent.h>
#include <string.h>
int main(void){
   struct dirent *files;
   DIR *test_dir;
   DIR *dir = opendir("Test");
   if (dir == NULL){
      printf("Directory cannot be opened!" );
      return 0;
   }
   printf("The files in folder are :\n");
   char fn[255];
   while ((files = readdir(dir)) != NULL)
   {
     strcpy( fn, "Test/" );
     strcat( fn, files->d_name );
     test_dir = opendir(fn);
     if (test_dir == NULL)
     {
        printf("%s\n", files->d_name);
     }
     closedir(test_dir);
   }
   closedir(dir);
   dir = opendir("Test");
   if (dir == NULL){
      printf("Directory cannot be opened!" );
      return 0;
   }
   printf("\n\nThe sub-folders in folder are :\n");
   while ((files = readdir(dir)) != NULL)
   {
     if ((strcmp( ".", files->d_name )) && (strcmp( "..", files->d_name )))
     {
       strcpy( fn, "Test/" );
       strcat( fn, files->d_name );
       test_dir = opendir(fn);
       if (test_dir != NULL)
       {
          printf("%s\n", files->d_name);
       }
       closedir(test_dir);
     }
   }
   closedir(dir);
   return 0;
}

int main()
{
    FILE *source1, *source2;
    char ch;

    source1 = fopen("s1.txt", "r+");
    source2 = fopen("s2.txt", "w");
    ch = getc(source1);
    while (ch != EOF)
    {
        ch = getc(source1);
    }
    ch = getc(source2);
    while (ch != EOF)
    {
        putc(ch,source1)
        ch = getc(source2);
    }
    fclose(source1);
    fclose(source2);
    printf("Appending completed .... ");
    return 0;
}

  1. Write a C program to insert 5 line to a file then use a function insert line number at the beginning of each line.

Answer

#include<stdio.h>
#include<stdlib.h>
#include<string.h>
int main()
{
  FILE *fp, *fp1, *fp2;
  int c=0, line_number=1;
  char ch, *line, *new_line, sep[3]=". \0";
  fp = fopen ("dd.txt", "r");
  fp1 = fopen ("dd.txt", "r");
  fp2 = fopen ("dd_new.txt", "w");
  ch = getc(fp);
  while (ch != EOF)
  {
    if (ch == '\n')
    {
      line = (char *)malloc((c+3)*sizeof(char));
      new_line = (char *)malloc((c+10)*sizeof(char));
      sprintf(new_line,"%d" ,line_number);
      fgets(line, c+3, fp1);
      strcat( new_line, sep);
      strcat( new_line, line);
      fputs(new_line, fp2);
      printf("%s",new_line);
      line_number=line_number+1;
      c=0;
    }
    else
    {
      c=c+1;
    }
    ch = getc(fp);
  }
  fclose(fp);
  fclose(fp1);
  fclose(fp2);
  fp = fopen ("dd.txt", "w");
  fp2 = fopen ("dd_new.txt", "r");
  ch = getc(fp2);
  while (ch != EOF)
    {
        putc(ch, fp);
        ch = getc(fp2);
    }
    fclose(fp);
    fclose(fp2);
  remove("dd_new.txt")
  return(0);
}
